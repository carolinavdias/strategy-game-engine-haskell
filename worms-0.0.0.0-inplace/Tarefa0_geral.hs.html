   19 -- | Uma posição numa matriz é dada como um par (/linha/,/colunha/).
   20 -- As coordenadas são dois números naturais e começam com (0,0) no canto superior esquerdo, com as linhas incrementando para baixo e as colunas incrementando para a direita.

   36 -- | Verifica se o indice pertence à lista.
   37 eIndiceListaValido :: Int -> [a] -> Bool
   38 eIndiceListaValido i l = i>=0 && i < length l

   40 -- | Calcula a dimensão de uma matriz.
   41 --
   42 -- __NB:__ Note que não existem matrizes de dimensão /m * 0/ ou /0 * n/, e que qualquer matriz vazia deve ter dimensão /0 * 0/.
   43 dimensaoMatriz :: Matriz a -> Dimensao
   44 dimensaoMatriz [] = (0,0)

   45 dimensaoMatriz m = (length m ,length (head m))

   47 -- | Verifica se a posição pertence à matriz.
   48 ePosicaoMatrizValida :: Posicao -> Matriz a -> Bool 
   49 ePosicaoMatrizValida (l, c) m = l >= 0 && l < length m && c>= 0 && c < length (head m)

   51 -- | Mover uma posição uma unidade no sentido de uma direção.
   52 movePosicao :: Direcao -> Posicao -> Posicao
   53 movePosicao Norte (l, c) = (l-1, c)

   54 movePosicao Sul (l, c) = (l+1,c)

   55 movePosicao Este (l, c) = (l, c+1)

   56 movePosicao Oeste (l, c) = (l, c-1)

   57 movePosicao Nordeste (l, c) = (l-1, c+1)

   58 movePosicao Sudeste (l, c) = (l+1, c+1)

   59 movePosicao Noroeste (l, c) = (l-1, c-1)

   60 movePosicao Sudoeste (l, c) = (l +1, c-1)
   61 
   62 -- | Versão da função 'movePosicao' que garante que o movimento não se desloca para fora de uma janela.
   63 --
   64 -- __NB:__ Considere uma janela retangular com origem no canto superior esquerdo definida como uma matriz. A função recebe a dimensao da janela.
   65 movePosicaoJanela :: Dimensao -> Direcao -> Posicao -> Posicao
   66 movePosicaoJanela (maxL, maxC) dir pos = 

   67     let novaPosicao = movePosicao dir pos

   68         (l, c) = novaPosicao

   69     in if l >= 0 && l < maxL && c >= 0 && c<maxC

   70         then  novaPosicao

   71         else pos
   72 
   73 -- n sei se é menor e igua ou só menor e n sei o que por no else
   74 
   75 -- | Converte uma posição no referencial em que a origem é no canto superior esquerdo da janela numa posição em que a origem passa a estar no centro da janela.
   76 --
   77 -- __NB:__ Considere posições válidas. Efetue arredondamentos como achar necessário. 
   78 
   79 origemAoCentro :: Dimensao -> Posicao -> Posicao
   80 origemAoCentro (lt,ct) (l, c) = (l - div lt 2, c-div ct 2)

   83 -- | Roda um par (posição,direção) 25% para a direita.
   84 --
   85 -- __NB:__ Vendo um par (posição,direção) como um vector, avança da posição de origem 
   86 -- para a posição de destino do vetor, e cria um novo vetor a partir da 
   87 -- posição de destino com a próxima direção na rosa dos ventos para a direita.
   88 --
   89 -- <<https://haslab.github.io/Teaching/LI1/2526/img/rodaposicaodirecao.png>>
   90 
   91 rodaPosicaoDirecao :: (Posicao,Direcao) -> (Posicao,Direcao)
   92 rodaPosicaoDirecao (pos, Noroeste) = (movePosicao Noroeste pos, Norte)

   93 rodaPosicaoDirecao (pos, dir) = (movePosicao dir pos, succ dir)

   97 -- | Devolve o elemento num dado índce de uma lista.
   98 --
   99 -- __NB:__ Retorna @Nothing@ se o índice não existir.
  100 encontraIndiceLista :: Int -> [a] -> Maybe a
  101 encontraIndiceLista _ [] = Nothing

  102 encontraIndiceLista 0 (x:_) = Just x

  103 encontraIndiceLista a (_:xs) = encontraIndiceLista (a-1) xs
  104 
  105 -- | Modifica um elemento num dado índice.
  106 --
  107 -- __NB:__ Devolve a própria lista se o elemento não existir.
  108 atualizaIndiceLista :: Int -> a -> [a] -> [a]
  109 atualizaIndiceLista _ _ [] = []

  110 atualizaIndiceLista 0 novo (_:xs) = novo:xs

  111 atualizaIndiceLista i novo (x:xs) = x : atualizaIndiceLista (i-1) novo xs
  112 
  113 -- | Devolve o elemento numa dada posição de uma matriz.
  114 --
  115 -- __NB:__ Retorna @Nothing@ se a posição não existir.
  116 encontraPosicaoMatriz :: Posicao -> Matriz a -> Maybe a
  117 encontraPosicaoMatriz (l, c) m 

  118     | l < 0 || c < 0 = Nothing

  119     | l >= length m = Nothing

  120     | c >= length (head m) = Nothing

  121     | otherwise = Just ((m !! l) !! c)
  122

  125 -- | Modifica um elemento numa dada posição de uma matriz.
  126 --
  127 -- __NB:__ Devolve a própria matriz se o elemento não existir.
  128 atualizaPosicaoMatriz :: Posicao -> a -> Matriz a -> Matriz a
  129 atualizaPosicaoMatriz (l, c) n x 

  130     | l < 0 || c < 0 = x

  131     | l >= length x = x

  132     | c >= length (head x) = x

  133     |otherwise = atualizaIndiceLista l (atualizaIndiceLista c n (x !! l)) x

  135 -- | Aplica uma sequência de movimentações a uma posição, pela ordem em que ocorrem na lista.
  136 moveDirecoesPosicao :: [Direcao] -> Posicao -> Posicao
  137 moveDirecoesPosicao [] x = x

  138 moveDirecoesPosicao (dir:dirs) (l, c) = moveDirecoesPosicao dirs (movePosicao dir (l,c))
  139 
  140 -- | Aplica a mesma movimentação a uma lista de posições.
  141 moveDirecaoPosicoes :: Direcao -> [Posicao] -> [Posicao]
  142 moveDirecaoPosicoes _ [] = []

  143 moveDirecaoPosicoes dir (x:xs) = movePosicao dir x : moveDirecaoPosicoes dir xs

  145 -- | Verifica se uma matriz é válida, no sentido em que modela um rectângulo.
  146 --
  147 -- __NB:__ Todas as linhas devem ter o mesmo número de colunas. 
  148 eMatrizValida :: Matriz a -> Bool
  149 eMatrizValida [] = True

  150 eMatrizValida m = all (\linha -> length linha == length (head m)) m
