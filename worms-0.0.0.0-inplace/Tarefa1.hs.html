   11 -- | Valida se um estado é válido segundo as regras do jogo.
   12 validaEstado :: Estado -> Bool
   13 validaEstado estado = 

   14     validaMapa (mapaEstado estado) &&

   15     validaObjetos (objetosEstado estado) (mapaEstado estado) (minhocasEstado estado) &&

   16     validaMinhocas (minhocasEstado estado) (mapaEstado estado) (objetosEstado estado)

   18 -- ========== VALIDAÇÃO DE MAPA ==========
   19 
   20 -- | O mapa é válido quando:
   21 -- 1. Não vazio
   22 -- 2. Denota corretamente uma grelha (mesmo número de colunas em todas as linhas)
   23 -- 3. Contém terrenos válidos (Ar/Agua/Terra/Pedra)
   24 
   25 validaMapa :: Mapa -> Bool
   26 validaMapa mapa =

   27     not (null mapa) && 

   28     eMatrizValida mapa && 

   29     all validaTerreno (concat mapa)

   30   where

   31     validaTerreno :: Terreno -> Bool

   32     validaTerreno Ar = True

   33     validaTerreno Agua = True

   34     validaTerreno Terra = True

   35     validaTerreno Pedra = True
   36 
   37 -- ========== VALIDAÇÃO DE OBJETOS ==========

   39 -- | Valida todos os objetos do estado.
   40 validaObjetos :: [Objeto] -> Mapa -> [Minhoca] -> Bool
   41 validaObjetos objs mapa minhocas = 

   42     all (\obj -> validaObjeto obj mapa minhocas objs) objs &&

   43     validaDisparosUnicos objs
   44 
   45 -- | Valida um objeto individual.
   46 validaObjeto :: Objeto -> Mapa -> [Minhoca] -> [Objeto] -> Bool
   47 validaObjeto obj mapa minhocas todosObjs = 

   48     case obj of

   49         Barril pos _ -> 

   50             posicaoValidaELivreNoMapa pos mapa &&

   51             posicaoNaoOcupadaPorOutroBarril pos todosObjs obj &&

   52             posicaoNaoOcupadaPorMinhoca pos minhocas

   53         

   54         Disparo pos dir tipo tempo dono -> 

   55             validaDisparo pos dir tipo tempo dono mapa minhocas
   56 
   57 -- | Valida um disparo.
   58 validaDisparo :: Posicao -> Direcao -> TipoArma -> Maybe Ticks -> NumMinhoca -> Mapa -> [Minhoca] -> Bool
   59 validaDisparo pos dir tipo tempo dono mapa minhocas =

   60     tipo /= Jetpack &&

   61     tipo /= Escavadora &&

   62     validaPosicaoDisparo pos dir tipo mapa &&

   63     validaTempoDisparo tipo tempo &&

   64     donoValido dono minhocas
   65 
   66 -- | Verifica se a posição é válida e livre no mapa (não contém terreno opaco).
   67 posicaoValidaELivreNoMapa :: Posicao -> Mapa -> Bool
   68 posicaoValidaELivreNoMapa pos mapa = 

   69     ePosicaoMatrizValida pos mapa &&

   70     case encontraPosicaoMatriz pos mapa of

   71         Just terreno -> not (eTerrenoOpaco terreno)

   72         Nothing -> False
   73 
   74 -- | Valida posição de disparo.
   75 -- Bazuca pode perfurar terreno opaco apenas na superfície.
   76 validaPosicaoDisparo :: Posicao -> Direcao -> TipoArma -> Mapa -> Bool
   77 validaPosicaoDisparo pos dir Bazuca mapa = 

   78     ePosicaoMatrizValida pos mapa &&

   79     case encontraPosicaoMatriz pos mapa of

   80         Just terreno -> 

   81             if eTerrenoOpaco terreno

   82             then posicaoAnteriorNaoOpaca pos dir mapa

   83             else True

   84         Nothing -> False

   85 validaPosicaoDisparo pos _ _ mapa = posicaoValidaELivreNoMapa pos mapa
   86 
   87 -- | Verifica se a posição anterior (direção oposta) não é opaca.
   88 posicaoAnteriorNaoOpaca :: Posicao -> Direcao -> Mapa -> Bool
   89 posicaoAnteriorNaoOpaca pos dir mapa = 

   90     let posAnt = movePosicao (direcaoOposta dir) pos

   91     in case encontraPosicaoMatriz posAnt mapa of

   92         Just terreno -> not (eTerrenoOpaco terreno)

   93         Nothing -> True
   94 
   95 -- | Retorna a direção oposta.
   96 direcaoOposta :: Direcao -> Direcao
   97 direcaoOposta Norte = Sul

   98 direcaoOposta Sul = Norte

   99 direcaoOposta Este = Oeste

  100 direcaoOposta Oeste = Este

  101 direcaoOposta Nordeste = Sudoeste

  102 direcaoOposta Sudoeste = Nordeste

  103 direcaoOposta Noroeste = Sudeste

  104 direcaoOposta Sudeste = Noroeste

  106 -- | Verifica se a posição não está ocupada por outro barril.
  107 posicaoNaoOcupadaPorOutroBarril :: Posicao -> [Objeto] -> Objeto -> Bool
  108 posicaoNaoOcupadaPorOutroBarril pos objs objAtual = 

  109     all (\obj -> case obj of

  110             Barril p _ -> p /= pos || objetosIguais obj objAtual

  111             _ -> True) objs
  112 
  113 -- | Verifica se dois objetos são o mesmo (para evitar auto-comparação).
  114 objetosIguais :: Objeto -> Objeto -> Bool
  115 objetosIguais (Barril p1 e1) (Barril p2 e2) = p1 == p2 && e1 == e2

  116 objetosIguais _ _ = False
  117 
  118 -- | Verifica se a posição não está ocupada por uma minhoca.
  119 posicaoNaoOcupadaPorMinhoca :: Posicao -> [Minhoca] -> Bool
  120 posicaoNaoOcupadaPorMinhoca pos minhocas = 

  121     all (\m -> posicaoMinhoca m /= Just pos) minhocas
  122 
  123 -- | Verifica se o dono é um índice válido na lista de minhocas.
  124 donoValido :: NumMinhoca -> [Minhoca] -> Bool
  125 donoValido num minhocas = num >= 0 && num < length minhocas
  126 
  127 -- | Valida o tempo do disparo conforme o tipo de arma.
  128 validaTempoDisparo :: TipoArma -> Maybe Ticks -> Bool
  129 validaTempoDisparo Bazuca Nothing = True

  130 validaTempoDisparo Bazuca (Just _) = False

  131 validaTempoDisparo Mina Nothing = True

  132 validaTempoDisparo Mina (Just t) = t >= 0 && t <= 2

  133 validaTempoDisparo Dinamite (Just t) = t >= 0 && t <= 4

  134 validaTempoDisparo Dinamite Nothing = False

  135 validaTempoDisparo _ _ = False
  136 
  137 -- | Valida que cada dono não tem disparos duplicados do mesmo tipo.
  138 validaDisparosUnicos :: [Objeto] -> Bool
  139 validaDisparosUnicos objs = 

  140     let disparos = [(tipo, dono) | Disparo _ _ tipo _ dono <- objs]

  141     in length disparos == length (removerDuplicados disparos)
  142 
  143 -- | Remove duplicados de uma lista.
  144 removerDuplicados :: Eq a => [a] -> [a]
  145 removerDuplicados [] = []

  146 removerDuplicados (x:xs) = x : removerDuplicados (filter (/= x) xs)

  148 -- ========== VALIDAÇÃO DE MINHOCAS ==========
  149 
  150 --Cada minhoca é válida, o que se verifica quando:
  151 --Tem uma posição válida e livre, ou opcionalmente nenhuma posição.
  152 --A sua posição não se encontra ocupada por um objeto de barril ou por outra minhoca.
  153 --Quando não tem posição ou se encontra numa posição em que o terreno é água, a minhoca tem que estar obrigatoriamente morta.
  154 --Quando viva, a vida da minhoca é um inteiro entre 0 e 100.
  155 --A quantidade de munições das diversas armas é um número inteiro maior ou igual a 0.
  156 
  157 -- | Valida todas as minhocas.
  158 validaMinhocas :: [Minhoca] -> Mapa -> [Objeto] -> Bool
  159 validaMinhocas minhocas mapa objs = 

  160     all (\m -> validaMinhoca m mapa objs minhocas) minhocas &&

  161     validaPosicoesMinhocasUnicas minhocas
  162 
  163 -- | Valida uma minhoca individual.
  164 validaMinhoca :: Minhoca -> Mapa -> [Objeto] -> [Minhoca] -> Bool
  165 validaMinhoca minhoca mapa objs todasMinhocas = 

  166     case posicaoMinhoca minhoca of

  167         Nothing -> 

  168             vidaMinhoca minhoca == Morta

  169         Just pos -> 

  170             posicaoValidaELivreNoMapa pos mapa &&

  171             posicaoLivreNoEstado pos objs todasMinhocas minhoca &&

  172             validaMinhocaEmAgua pos mapa minhoca &&

  173             validaVidaEMunicoes minhoca
  174 
  175 -- | Verifica se a posição está livre no estado (não tem barril nem outra minhoca).
  176 posicaoLivreNoEstado :: Posicao -> [Objeto] -> [Minhoca] -> Minhoca -> Bool
  177 posicaoLivreNoEstado pos objs minhocas minhocaAtual =

  178     posicaoNaoTemBarril pos objs &&

  179     posicaoNaoOcupadaPorOutraMinhoca pos minhocas minhocaAtual
  180 
  181 -- | Verifica se a posição não tem barril.
  182 posicaoNaoTemBarril :: Posicao -> [Objeto] -> Bool
  183 posicaoNaoTemBarril pos objs = 

  184     all (\obj -> case obj of

  185             Barril p _ -> p /= pos

  186             _ -> True) objs
  187 
  188 -- | Verifica se a posição não está ocupada por outra minhoca.
  189 posicaoNaoOcupadaPorOutraMinhoca :: Posicao -> [Minhoca] -> Minhoca -> Bool
  190 posicaoNaoOcupadaPorOutraMinhoca pos minhocas minhocaAtual = 

  191     all (\m -> posicaoMinhoca m /= Just pos || 

  192                minhocasIguais m minhocaAtual) minhocas
  193 
  194 -- | Compara minhocas (são iguais se tiverem a mesma posição e vida).
  195 minhocasIguais :: Minhoca -> Minhoca -> Bool
  196 minhocasIguais m1 m2 = 

  197     posicaoMinhoca m1 == posicaoMinhoca m2 &&

  198     vidaMinhoca m1 == vidaMinhoca m2
  199 
  200 -- | Verifica se minhoca em água está morta.
  201 validaMinhocaEmAgua :: Posicao -> Mapa -> Minhoca -> Bool
  202 validaMinhocaEmAgua pos mapa minhoca = 

  203     case encontraPosicaoMatriz pos mapa of

  204         Just Agua -> vidaMinhoca minhoca == Morta

  205         _ -> True
  206 
  207 -- | Valida vida e munições da minhoca.
  208 validaVidaEMunicoes :: Minhoca -> Bool
  209 validaVidaEMunicoes minhoca = 

  210     validaVida (vidaMinhoca minhoca) &&

  211     jetpackMinhoca minhoca >= 0 &&

  212     escavadoraMinhoca minhoca >= 0 &&

  213     bazucaMinhoca minhoca >= 0 &&

  214     minaMinhoca minhoca >= 0 &&

  215     dinamiteMinhoca minhoca >= 0

  216   where

  217     validaVida :: VidaMinhoca -> Bool

  218     validaVida Morta = True

  219     validaVida (Viva v) = v >= 0 && v <= 100
  220 
  221 -- | Valida que minhocas não compartilham a mesma posição.
  222 validaPosicoesMinhocasUnicas :: [Minhoca] -> Bool
  223 validaPosicoesMinhocasUnicas minhocas = 

  224     let posicoes = [pos | m <- minhocas, Just pos <- [posicaoMinhoca m]]

  225     in length posicoes == length (removerDuplicados posicoes)
