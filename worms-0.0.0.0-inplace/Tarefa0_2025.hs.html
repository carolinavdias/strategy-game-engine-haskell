   31 -- | Verifica se um tipo de terreno é destrutível, i.e., pode ser destruído por explosões.
   32 --
   33 -- __NB:__ Apenas @Terra@ é destrutível.
   34 eTerrenoDestrutivel :: Terreno -> Bool
   35 eTerrenoDestrutivel Terra = True

   36 eTerrenoDestrutivel _ = False
   37 
   38 -- | Verifica se um tipo de terreno é opaco, i.e., não permite que objetos ou minhocas se encontrem por cima dele.
   39 --
   40 -- __NB:__ Apenas @Terra@ ou @Pedra@ são opacos.
   41 eTerrenoOpaco :: Terreno -> Bool
   42 eTerrenoOpaco Terra = True

   43 eTerrenoOpaco Pedra = True

   44 eTerrenoOpaco _ = False
   45 
   46 -- | Verifica se uma posição do mapa está livre, i.e., pode ser ocupada por um objeto ou minhoca.
   47 --
   48 -- __NB:__ Uma posição está livre se não contiver um terreno opaco.
   49 ePosicaoMapaLivre :: Posicao -> Mapa -> Bool
   50 ePosicaoMapaLivre pos mapa = 

   51     case encontraPosicaoMatriz pos mapa of

   52         Just terreno -> not (eTerrenoOpaco terreno)

   53         Nothing -> False  -- se a pos não existe no mapa, não está livre!
   54 
   55 -- | Verifica se uma posição do estado está livre, i.e., pode ser ocupada por um objeto ou minhoca.
   56 --
   57 -- __NB:__ Uma posição está livre se o mapa estiver livre e se não estiver já uma minhoca ou um barril nessa posição.
   58 ePosicaoEstadoLivre :: Posicao -> Estado -> Bool
   59 ePosicaoEstadoLivre pos estado = 

   60     ePosicaoMapaLivre pos (mapaEstado estado) &&

   61     not (posicaoTemMinhoca pos (minhocasEstado estado)) &&

   62     not (posicaoTemBarril pos (objetosEstado estado))

   63   where

   64     -- verificar se há uma minhoca na posição:posicaoTemMinhoca

   65     posicaoTemMinhoca :: Posicao -> [Minhoca] -> Bool

   66     posicaoTemMinhoca _ [] = False

   67     posicaoTemMinhoca p (m:ms) = 

   68         case posicaoMinhoca m of

   69             Just posMinhoca -> p == posMinhoca || posicaoTemMinhoca p ms

   70             Nothing -> posicaoTemMinhoca p ms

   71     

   72     -- verifica se há um barril na posição: posicaoTemBarril

   73     posicaoTemBarril :: Posicao -> [Objeto] -> Bool

   74     posicaoTemBarril _ [] = False

   75     posicaoTemBarril p (obj:objs) = 

   76         case obj of

   77             Barril posBarril _ -> p == posBarril || posicaoTemBarril p objs

   78             _ -> posicaoTemBarril p objs

   80 -- | Verifica se numa lista de objetos já existe um disparo feito para uma dada arma por uma dada minhoca.
   81 minhocaTemDisparo :: TipoArma -> NumMinhoca -> [Objeto] -> Bool
   82 minhocaTemDisparo _ _ [] = False

   83 minhocaTemDisparo arma numMinhoca (obj:objs) =

   84     case obj of

   85         Disparo _ _ tipoDisp _ dono -> 

   86             (tipoDisp == arma && dono == numMinhoca) || minhocaTemDisparo arma numMinhoca objs

   87         _ -> minhocaTemDisparo arma numMinhoca objs

   89 -- | Destrói uma dada posição no mapa (tipicamente como consequência de uma explosão).
   90 --
   91 -- __NB__: Só terrenos @Terra@ pode ser destruídos.
   92 destroiPosicao :: Posicao -> Mapa -> Mapa
   93 destroiPosicao pos mapa = 

   94     case encontraPosicaoMatriz pos mapa of

   95         Just terreno -> 

   96             if eTerrenoDestrutivel terreno

   97                 then atualizaPosicaoMatriz pos Ar mapa

   98                 else mapa

   99         Nothing -> mapa  -- se a posição não existe, retorna o mapa não alterado
